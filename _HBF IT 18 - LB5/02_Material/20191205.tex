				int led = 9;
				int taster = 11;
				
				void setup() {
					pinMode(led,OUTPUT);
					pinMode(taster,INPUT_PULLUP);
					digitalWrite(taster,HIGH);
				}
				
				void loop() {
					int bRead = digitalRead(taster);
					if(bRead == LOW) {
						doThis();
						delay(1000);
					}
				}
				
				void doThis() {
					digitalWrite(led,HIGH);
					delay(2000);
					digitalWrite(led,LOW);
				}
